function int fiboR(int x) {
	if (x == 0) {
		return 0
	} elseif (x == 1){
		return 1
	}

	return fiboR(x-1) + fiboR(x-2);
}

function int fiboI(int n) {
	int a
	int b
	int i
	int aux

	a = 1
	b = 1
	i = 3

	while (i <= n) {
		aux = a + b
		a = b
		b = aux
		i = i + 1
	}

	return b
}

main{
	int x
	int y

	x = 10
	y = 10

	x = fiboR(x)
	print("Fibonacci rec: " + x)

	y = fiboI(y)
	print("Fibonacci iter: " + y)
}

end
