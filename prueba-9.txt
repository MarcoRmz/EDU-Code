function int factorialCic(int n){
  int i
  int res
  i = 1
  res = n
  while(i < n){
    res = i * res
    i = i + 1
  }
  return res
}

function int factorialRec(int n){
  if( n == 1){
    return 1
  }elseif(n > 1){
    return n * factorialRec(n-1)
  }
}


main{
int x
bool y
x = 12
print("Factorial rec of "+x+": "+factorialRec(x))
print("Factorial cic of "+x+": "+factorialCic(x))
y = factorialRec(x) == factorialCic(x)
print("Cic == Rec " + y)
}

end
